   - **CSS Scoping with `data-scope`**: Fix technique — inject a unique `data-scope="bsvg-{id}"` attribute on `<svg>` and prefix `.cls-*` rules with `[data-scope="bsvg-..."]` to isolate them, while preserving `html[lang]` rules globally for WPML

     private function scope_svg_styles( string $svg, string $scope_id ): string {
         $svg = preg_replace( '/<svg\b/', '<svg data-scope="' . esc_attr( $scope_id ) . '"', $svg, 1 );
         $svg = preg_replace_callback(
             '/(<style[^>]*>)(.*?)(<\/style>)/is',
             static function ( $m ) use ( $scope_id ) {
                 $scoped_css = preg_replace_callback(
                     '/([^{}]+)\{([^{}]*)\}/s',
                     static function ( $rule ) use ( $scope_id ) {
                         $selector = trim( $rule[1] );
                         if ( strncmp( $selector, 'html', 4 ) === 0 ) {
                             return $rule[0];
                         }
                         return '[data-scope="' . $scope_id . '"] ' . $selector . ' {' . $rule[2] . '}';
                     },
                     $m[2]
                 );
                 return $m[1] . $scoped_css . $m[3];
             },
             $svg
         );
         return $svg;
     }
     
     // In render(), after file_get_contents and XML cleanup:
     $svg = $this->scope_svg_styles( $svg, 'bsvg-' . $this->get_id() );

# Plano: Módulo Attack Monitoring — bit-monitoring

## Context

O site concertacaoamazonia.com.br sofreu um ataque ativo (CloudFront 5XX Rate 27.5%) via distribuição CloudFront de terceiros apontando para o ALB como origin. Identificou-se que:
- 38.419 bloqueios em 7 dias (CloudFront WAF — ACL-WPAdminHML)
- 17.406 bloqueios em 7 dias (ALB WAF — amazonia-waf, sem logging ativo)
- atacante usa `Apache-HttpClient/4.5.2 (Java/1.8.0_161)` via IPs CLOUDFRONT_ORIGIN_FACING
- O bit-monitoring (`~/scripts/bit-monitoring`) é uma plataforma FastAPI + Next.js 14 + Celery + PostgreSQL/TimescaleDB + Redis

**Objetivo:** Criar módulo `security_events` no bit-monitoring para consolidar e visualizar todos os ataques bloqueados pelos recursos AWS (WAF CloudFront, WAF ALB).

### Passo 3: IP Enrichment Service (`backend/app/services/ip_enrichment.py`)

Cache em **Redis** (não SQLite — sem lock contention entre workers Celery):

```python
import json, httpx, redis, time
from app.core.config import settings

class IpEnrichmentService:
    """ip-api.com: 45 req/min. Redis cache 7 dias. Multi-worker safe."""
    def __init__(self):
        self.redis = redis.from_url(settings.REDIS_URL)
        self.cache_ttl = 7 * 86400

    def enrich(self, ip: str) -> dict:
        cache_key = f"ip:enrichment:{ip}"
        cached = self.redis.get(cache_key)
        if cached:
            return json.loads(cached)
        try:
            resp = httpx.get(f"http://ip-api.com/json/{ip}",
                params={"fields": "country,countryCode,city,org,as,query"},
                timeout=3.0)
            data = resp.json()
            self.redis.setex(cache_key, self.cache_ttl, json.dumps(data))
            return data
        except Exception:
            return {}

    def enrich_batch(self, ips: list[str]) -> dict[str, dict]:
        """Respeita rate limit: lotes de 9 IPs a cada 12s (< 45/min)."""
        results = {}
        new_ips = [ip for ip in ips if not self.redis.get(f"ip:enrichment:{ip}")]
        for i in range(0, len(new_ips), 9):
            batch = new_ips[i:i+9]
            for ip in batch:
                results[ip] = self.enrich(ip)
            if i + 9 < len(new_ips):
                time.sleep(12)
        for ip in ips:
            if ip not in results:
                results[ip] = self.enrich(ip)
        return results
```

### Passo 4: WAF Collector Service (`backend/app/services/waf_collector.py`)

Usa **AWSClientFactory** (não boto3.Session — mantém credential caching do projeto):

```python
from datetime import datetime
from app.providers.aws import AWSClientFactory
from app.models.cloud_account import CloudAccount

class WafCollectorService:
    def __init__(self, cloud_account: CloudAccount):
        self.client_factory = AWSClientFactory(cloud_account)
        self.region = "sa-east-1"

    def query_cloudfront_waf(self, since_timestamp: datetime) -> list[dict]:
        return self._query_waf_logs("aws-waf-logs-concertacao", since_timestamp)

    def query_alb_waf(self, since_timestamp: datetime) -> list[dict]:
        return self._query_waf_logs("aws-waf-logs-amazonia-alb", since_timestamp)

    def _query_waf_logs(self, log_group: str, since_timestamp: datetime) -> list[dict]:
        """
        CloudWatch Logs Insights é ASYNC — start_query + polling obrigatório.
        Watermark cursor: busca desde since_timestamp (não janela fixa).
        """
        logs_client = self.client_factory.get_client("logs", region=self.region)
        query = """
        fields @timestamp, action, terminatingRuleId,
               httpRequest.clientIp, httpRequest.uri,
               httpRequest.headers.0.value as user_agent
        | filter action = 'BLOCK'
        | sort @timestamp asc
        | limit 10000
        """
        response = logs_client.start_query(
            logGroupName=log_group,
            startTime=int(since_timestamp.timestamp()),
            endTime=int(datetime.utcnow().timestamp()),
            queryString=query,
        )
        return self._poll_query(logs_client, response["queryId"])

    def _poll_query(self, client, query_id: str, timeout: int = 60) -> list[dict]:
        import time as t
        for _ in range(timeout):
            result = client.get_query_results(queryId=query_id)
            if result["status"] == "Complete":
                return result["results"]
            if result["status"] in ("Failed", "Cancelled"):
                return []
            t.sleep(1)
        return []
```

from app.models.cloud_account import CloudAccount

def _get_cloud_account(db) -> CloudAccount:
    """Busca a CloudAccount da Concertação (primary AWS account)."""
    from app.core.config import settings
    return db.execute(
        select(CloudAccount).where(CloudAccount.id == settings.CONCERTACAO_AWS_ACCOUNT_ID)
    ).scalar_one()

def _parse_events(raw_events: list, source: str) -> list[SecurityEvent]:
    """
    CloudWatch Logs Insights retorna: [{'field': '@timestamp', 'value': '...'}, ...]
    Converte para objetos SecurityEvent.
    """
    parsed = []
    for event in raw_events:
        fields = {f["field"]: f["value"] for f in event}
        try:
            parsed.append(SecurityEvent(
                timestamp=datetime.fromisoformat(fields.get("@timestamp", "").replace("Z", "+00:00")),
                source=source,
                rule_name=fields.get("terminatingRuleId"),
                action=fields.get("action", "BLOCK"),
                ip_address=fields.get("httpRequest.clientIp"),
                user_agent=fields.get("user_agent"),
                uri=fields.get("httpRequest.uri"),
            ))
        except (ValueError, KeyError):
            continue
    return parsed

@celery_app.task(name="workers.tasks.waf_collector.collect_waf_events")
def collect_waf_events():
    """
    A cada 5 min. Watermark cursor: busca apenas eventos NOVOS desde
    último timestamp processado por source (não janela fixa de 6h).
    Upsert com on_conflict_do_nothing para deduplicação.
    """
    with sync_db_session() as db:
        collector = WafCollectorService(_get_cloud_account(db))
        enricher = IpEnrichmentService()

        for source, fetch_fn in [
            ("waf_cloudfront", collector.query_cloudfront_waf),
            ("waf_alb", collector.query_alb_waf),
        ]:
            last_ts = db.execute(
                select(func.max(SecurityEvent.timestamp)).where(
                    SecurityEvent.source == source
                )
            ).scalar() or (datetime.utcnow() - timedelta(hours=6))

            raw = fetch_fn(since_timestamp=last_ts)
            if not raw:
                continue

            parsed = _parse_events(raw, source)
            unique_ips = list({e.ip_address for e in parsed if e.ip_address})
            enrichments = enricher.enrich_batch(unique_ips)

            for event in parsed:
                if event.ip_address and event.ip_address in enrichments:
                    geo = enrichments[event.ip_address]
                    event.ip_org = geo.get("org")
                    event.ip_city = geo.get("city")
                    event.ip_asn = geo.get("as")

            # Upsert: on_conflict_do_nothing evita duplicatas
            if parsed:
                stmt = pg_insert(SecurityEvent).values(
                    [{"timestamp": e.timestamp, "source": e.source,
                      "rule_name": e.rule_name, "action": e.action,
                      "ip_address": e.ip_address, "user_agent": e.user_agent,
                      "uri": e.uri, "ip_org": e.ip_org, "ip_city": e.ip_city,
                      "ip_asn": e.ip_asn}
                     for e in parsed]
                ).on_conflict_do_nothing()
                db.execute(stmt)
        db.commit()

def upgrade() -> None:
    # 1. Habilitar TimescaleDB (OBRIGATÓRIO antes de create_hypertable)
    op.execute("CREATE EXTENSION IF NOT EXISTS timescaledb;")

    # 2. Criar tabela
    op.create_table(
        'security_events',
        sa.Column('id', sa.Integer(), primary_key=True),
        sa.Column('timestamp', sa.DateTime(timezone=True), nullable=False),
        sa.Column('source', sa.String(32)),
        sa.Column('rule_name', sa.String(128)),
        sa.Column('action', sa.String(16)),
        sa.Column('country_code', sa.String(2)),
        sa.Column('ip_address', sa.String(45)),
        sa.Column('user_agent', sa.String(512)),
        sa.Column('uri', sa.String(1024)),
        sa.Column('http_method', sa.String(8)),
        sa.Column('status_code', sa.Integer()),
        sa.Column('bot_category', sa.String(64)),
        sa.Column('ip_org', sa.String(128)),
        sa.Column('ip_city', sa.String(64)),
        sa.Column('ip_asn', sa.String(64)),
        sa.Column('threat_score', sa.Float()),
    )
    op.create_index('ix_se_timestamp', 'security_events', ['timestamp'])
    op.create_index('ix_se_source_ts', 'security_events', ['source', 'timestamp'])
    op.create_index('ix_se_ip_ts', 'security_events', ['ip_address', 'timestamp'])

    # 3. Hypertable (1 dia por chunk — adequado para ~5k eventos/dia)
    op.execute("""
        SELECT create_hypertable('security_events', 'timestamp',
            chunk_time_interval => INTERVAL '1 day',
            if_not_exists => TRUE)
    """)

    # 4. Compressão automática após 7 dias (~60% de economia de storage)
    op.execute("""
        SELECT add_compression_policy('security_events',
            INTERVAL '7 days', if_not_exists => true)
    """)

    # 5. Retenção: 90 dias (compliance)
    op.execute("""
        SELECT add_retention_policy('security_events',
            INTERVAL '90 days', if_not_exists => true)
    """)

    # 6. Continuous aggregate para timeline (frontend consulta esta view)
    try:
        op.execute("""
            CREATE MATERIALIZED VIEW IF NOT EXISTS attack_timeline_1h
            WITH (timescaledb.continuous) AS
            SELECT time_bucket('1 hour', timestamp) AS hour,
                   source, action, count(*) AS block_count
            FROM security_events
            GROUP BY hour, source, action;
        """)
        op.execute("""
            SELECT add_continuous_aggregate_policy(
                'attack_timeline_1h',
                start_offset => INTERVAL '3 hours',
                end_offset => INTERVAL '0 seconds',
                schedule_interval => INTERVAL '5 minutes',
                if_not_exists => true)
        """)
    except Exception:
        # Fallback: view regular (sem continuous aggregate)
        op.execute("""
            CREATE OR REPLACE VIEW attack_timeline_1h AS
            SELECT date_trunc('hour', timestamp) AS hour,
                   source, action, count(*) AS block_count
            FROM security_events
            GROUP BY hour, source, action;
        """)


def downgrade() -> None:
    # Remover políticas antes de dropar (evita órfãos)
    for stmt in [
        "SELECT remove_continuous_aggregate_policy('attack_timeline_1h', if_exists => true)",
        "SELECT remove_retention_policy('security_events', if_exists => true)",
        "SELECT remove_compression_policy('security_events', if_exists => true)",
    ]:
        try:
            op.execute(stmt)
        except Exception:
            pass
    op.execute("DROP MATERIALIZED VIEW IF EXISTS attack_timeline_1h CASCADE")
    op.execute("DROP VIEW IF EXISTS attack_timeline_1h CASCADE")
